    public allDocs(options: Object): Promise <Response, Error> source

    Params:

    NameTypeAttributeDescription
    options Object

    The options object
    options.include_docs Boolean

    Include the document itself in each row in the doc field. Otherwise by default you only get the _id and _rev properties.
    options.conflicts Boolean

    Include conflict information in the _conflicts field of a doc.
    options.attachments Boolean

    Include attachment data as base64-encoded string.
    options.startkey String

    Get documents with IDs in range.
    options.endkey String

    Get documents with IDs in a certain range (inclusive/inclusive).
    options.inclusive_end Boolean

    Include documents having an ID equal to the given options.endkey. Default: true.
    options.limit Number

    Maximum number of documents to return.
    options.skip Number

    Number of docs to skip before returning (warning: poor performance on IndexedDB/LevelDB!).
    options.descending Boolean

    Reverse the order of the output documents.
    options.key String

    Only return documents with IDs matching this string key.
    options.keys Array

    Array of string keys to fetch in a single shot.

    Return:

    Promise <Response, Error>

    A promise that is resolved/rejected upon success/failure.

    Example:

    db.allDocs({
		limit: 5,
		include_docs: true
}).then(function(resp) {
		expect(resp.status).toBe(200);
		expect(resp.data.rows).toBeDefined();
		expect(resp.data.rows.length).toBe(5);
}).catch(function (err) {
		console.log(err);
});

    public changes(options: Object): Object source

    Params:

    NameTypeAttributeDescription
    options Object

    The options to send with the changes request. All options default to false unless otherwise specified.
    options.live Boolean

    Will emit change events for all future changes until cancelled.
    options.include_docs Boolean

    Include the associated document with each change.
    options.conflicts Boolean

    Include conflicts.
    options.attachments Boolean

    Include attachments.
    options.binary Boolean

    Return attachment data as Blobs/Buffers, instead of as base64-encoded strings.
    options.descending Boolean

    Reverse the order of the output documents.
    options.since Number

    Start the results from the change immediately after the given sequence number. You can also pass 'now' if you want only new changes (when live is true).
    options.limit Number

    Limit the number of results to this number.
    options.timeout Number

    Request timeout (in milliseconds).

    Return:

    Object

    An object with cancel() on(event) methods.

    Example:

    var db = new px.mobile.DB('testdb', {baseUrl: 'adapter://localhost:5984/default'});
var changes = db.changes({
		since: 'now',
		live: true,
		include_docs: true
})
.on('change', function(change) {
		console.warn('Change', change);
})
.on('complete', function(info) {
		console.warn('Changes completed', info);
})
.on('error', function (err) {
		console.log(err);
});
changes.cancel(); // whenever you want to cancel
